Basic Insertion Commands

(a b (|) c d)

)

(a b |c )
(a b c)|

paredit-open-square 

(a b |c d)
(a b [|] c d)

(foo "bar |baz" quux)
(foo "bar [|baz" quux)

]

paredit-close-round 

(define-key keymap [frob| ] \'frobnicate)

paredit-doublequote 

(frob grovel |full lexical)
(frob grovel "|" full lexical)
